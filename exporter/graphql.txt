query ($owner: String!, $repo: String!, $stars: Boolean!, $issues: Boolean!, $watchers: Boolean!, $forks: Boolean!) {
  repository(owner: $owner, name: $repo) {
    stargazers @include(if: $stars) {
      totalCount
    }
    issues @include(if: $issues) {
      totalCount
    }
    watchers @include(if: $watchers) {
      totalCount
    }
    forkCount @include(if: $forks)
  }
  rateLimit {
    limit
    cost
    remaining
    resetAt
  }
}
